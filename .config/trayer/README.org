* Killing
#+begin_src shell
if [[ $(pidof trayer) ]]
then
    killall trayer
fi
#+end_src

* Config
#+begin_src shell
sleep 2
trayer \
    --edge top \
    --align right \
    --widthtype request \
    --height $bar_size \
    --SetDockType true \
    --SetPartialStrut true \
    --transparent true \
    --tint $color \
    --alpha 0 \
    --expand true \
    --padding 4 \
    --monitor 1
    # --distancefrom right \
    # --distance 400 \
#+end_src
