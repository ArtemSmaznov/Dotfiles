* Size
#+begin_src jsonc
{
    "layer": "top",    // Waybar at top layer
    "position": "top", // Waybar position (top|bottom|left|right)
    "height": 24,      // Waybar height (to be removed for auto height)
    "spacing": 4,      // Gaps between modules (4px)
#+end_src
